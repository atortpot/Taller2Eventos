package com.example.taller2eventos;

import android.os.AsyncTask;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ProgressBar;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

public class PrincipalActivity extends AppCompatActivity {
    private EditText nombreEditText;
    private Button guardarButton, iniciarTareaButton;
    private TextView nombreTextView;
    private ProgressBar progressBar;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_principal);

        nombreEditText = findViewById(R.id.nombreEditText);
        guardarButton = findViewById(R.id.guardarButton);
        iniciarTareaButton = findViewById(R.id.iniciarTareaButton);
        nombreTextView = findViewById(R.id.nombreTextView);
        progressBar = findViewById(R.id.progressBar);

        // Guardar nombre
        guardarButton.setOnClickListener(v -> {
            String nombre = nombreEditText.getText().toString();
            nombreTextView.setText(nombre);
        });

        // Iniciar tarea en segundo plano
        iniciarTareaButton.setOnClickListener(v -> {
            new TareaEnSegundoPlano().execute();
        });
    }

    // AsyncTask para simular operación de red
    private class TareaEnSegundoPlano extends AsyncTask<Void, Integer, Void> {
        @Override
        protected void onPreExecute() {
            progressBar.setVisibility(View.VISIBLE);
        }

        @Override
        protected Void doInBackground(Void... voids) {
            for (int i = 1; i <= 10; i++) {
                try {
                    Thread.sleep(500); // Simula la operación de red
                    publishProgress(i * 10);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            return null;
        }

        @Override
        protected void onProgressUpdate(Integer... values) {
            progressBar.setProgress(values[0]);
        }

        @Override
        protected void onPostExecute(Void aVoid) {
            progressBar.setVisibility(View.GONE);
        }
    }
}
