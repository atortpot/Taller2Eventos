package com.example.taller2eventos;

import android.content.Intent;
import android.graphics.Color;
import android.os.Bundle;
import android.widget.Button;
import android.widget.LinearLayout;

import androidx.appcompat.app.AppCompatActivity;

public class ConfiguracionActivity extends AppCompatActivity {
    private LinearLayout layout;
    private Button btnCambiarColor, btnVolverInicio;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_configuracion);

        layout = findViewById(R.id.configLayout);
        btnCambiarColor = findViewById(R.id.btnCambiarColor);
        btnVolverInicio = findViewById(R.id.btnVolverInicio);

        // Cambiar color de fondo
        btnCambiarColor.setOnClickListener(v -> {
            layout.setBackgroundColor(Color.rgb((int) (Math.random() * 256),
                    (int) (Math.random() * 256),
                    (int) (Math.random() * 256)));
        });

        // Volver a la pantalla de inicio
        btnVolverInicio.setOnClickListener(v -> {
            Intent intent = new Intent(ConfiguracionActivity.this, MainActivity.class);
            startActivity(intent);
        });
    }
}
